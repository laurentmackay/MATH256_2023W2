\subsection{Preamble}

\slide[Recall:  Homogeneous Systems of $n$ 1$^{st}$ order ODEs]{\vspace{-1.5em}
\[ \dd{}{t}\vec{x} = \mathbf{A}(t) \vec{x}\] 
has $n$ linearly independent \alert{fundamental solutions} \vspace{-.5em} \[ \left\{ \vec{x}_1(t), \;\vec{x}_2(t),\;\dots, \;\vec{x}_n(t) \right\} \]
and its general solution is given by \vspace{-0.5em} \[\vec{x}(t) = c_1  \vec{x}_1(t)  +c_2 \vec{x}_2(t)  + \cdots + c_n\vec{x}_n(t) \]
 \vspace{-2.5em}
 
 or 
 
 \vspace{-2.5em}
 \[\vec{x} = \mathbf{X}(t) \vec{c} \;\;\text{where}\;\; \mathbf{X}(t) =\mat{cccc}{\vec{x}_1(t)  & \vec{x}_2(t)  & \cdots &\vec{x}_n(t) } \;\; \student{\text{and } \dd{}{t}\mathbf{X} = \mathbf{AX}} \]
\vspace{-.5em}

\subsection{Variation of Parameters}
\slide[Variation of Parameters: $\dd{}{t}\vec{x} = \mathbf{A}(t) \vec{x} + \vec{f}(t)$ ]{\vspace{-1em}
Due to the linearity of the DE, we know the solution structure is\vspace{-0.5em}
\[\vec{x}(t) =\underbrace{\vec{x}_h(t)}_{\text{homog. part}}+\underbrace{\vec{x}_p(t)}_{\text{particular part}} \quad \text{where}\;\;\vec{x}_h = \mathbf{X}(t) \vec{c}\quad \text{and}\quad \dd{}{t}\mathbf{X} =\mathbf{A}\mathbf{X}  \]\vspace{-1.25em}
\student{
\algn{&&\text{LHS:}\quad\dd{}{t}\vec{x}_p =  \mathbf{X}'\vec{u}+\mathbf{X}\vec{u}' \\
\text{\uline{Guess:} }\vec{x}_p&= \mathbf{X}(t) \vec{u}(t)&\phantom{\text{LHS:}\quad\dd{}{t}\vec{x}}=\mathbf{A}\mathbf{X}\vec{u}+\mathbf{X}\vec{u}' \\
\cancel{\mathbf{A}\mathbf{X}\vec{u}}+\mathbf{X}\vec{u}' &= \cancel{\mathbf{A}\mathbf{X}\vec{u}}+\vec{f}(t)&\text{RHS:}\quad\phantom{\dd{}{t}\vec{x}}\phantom{=}\mathbf{A}\mathbf{X}\vec{u}+\vec{f}(t)\\\\
\vec{u}' &= \mathbf{X}^{-1}\vec{f}(t) =\vec{g}(t) &\text{where $\vec{g}$ solves }\mathbf{X}\vec{g}=\vec{f}(t)\\\\
\vec{u}(t) &= \int \underbrace{ \mathbf{X}^{-1}(t)\vec{f}(t)}_{\vec{g}(t)} dt \Aboxed{\vec{x}_p&=\mathbf{X}(t) \int \mathbf{X}^{-1}(t)\vec{f}(t) dt }}
}
}
