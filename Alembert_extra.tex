\subsection{D'Alembert extra problems}
\slide[]{
Given\vspace{-2.25em}
\algn{y_{tt}&=25y_{xx}  & y_x(0,t)&=y_x(3,t)=0, \\   y(x,0)&=x^3-\frac92x^2+7 \quad \text{for }0<x<3& y_t(x,0)&=0}
\vfill
Find the solution $y(1,1000)$.

\student{\vfill
According to D'Alembert's solution method we have 
\algn{y&=\frac{F(1-5000)+F(1+5000)}{2}\intertext{where $F$ is an \uline{even} periodic extension of the initial condition}
F(z)&=\begin{cases} z^3-\frac92z^2+7& 0<z<3\\ -z^3-\frac92z^2+7  & -3<z<0\end{cases} & F(z+n6)=F(z) \quad \forall n\in\mathbb{N}}\vfill
The difficulty comes from the fact that $1-5000$ and $1+5000$ are not in the interval $[-3,3]$
}

}

\slide[]{
\student{
We must exploit the property $F(z+n6)=F(z)$  to map -4999 and 5001 onto $[-3,3]$
\vfill
Need to find two  appropriate integers $n_1$ and $n_2$
\vfill
\algn{-3&<-4999+n_16<3 & -3&<5001+n_26<3\\
4996&<n_16<5002 & -5004&<n_26<-4998\\
\frac{4996}{6}&<n_1<\frac{5002}{6} & \frac{-5004}{6}&<n_2<\frac{-4998}{6}\\
832.66&<n_1<833.6 & -833&<n_2<-834\\
n_1&=833 & n_2&=-834  \;(\text{or} -833)\intertext{So,}
-4999 &\rightarrow -4999+833\cdot6 = -1 & 5001 &\rightarrow 5001-834\cdot6=-3 \intertext{Finally, }}\vspace{-2.5em}
\[y(1,1000) =\frac{F(-1)+F(-3)}{2} =\frac{1-\frac92+7 + 27 -\frac92 \cdot 9 +7}{2} =\frac{-3}{2}\]


}
}

\slide{

\student{We can also use the solution from separation of variables
\vfill
\algn{y(1,1000) &= \frac{a_0}{2}+ \sum_{n=1}^\infty  a_n \cos \left( \frac{n \pi}{3} \right) \cos\left( n\pi \frac{5000}{3} \right) \intertext{with}
a_0&=\frac{2}{3}\int_0^3 x^3 -\frac92 x^2 +7 dx =\frac12 \\
a_n&=\frac{2}{3}\int_0^3 \left(x^3 -\frac92 x^2 +7 \right) \cdot \cos \left( \frac{n \pi}{3} x \right) dx = -\frac{324 \left((-1)^n-1\right)}{\pi ^4 n^4}}
\vfill
This is somewhat useless as we need to compute an infinite sum to obtain an actual value.
}
}
