\slide[Find the second solution to\hfill  $t^2y''+3ty'+y=0$\hfill]{\vspace{-1em}
Guess: $y_2=u(t)y_1$ where $u(t)$ is unknown and $y_1=t^{-1}$
\student{\[y_2' = u't^{-1}-ut^{-2},\qquad y_2''=u''t^{-1}-u't^{-2}-u't^{-2}+2ut^{-3} \]
Plug into DE:
\algn{t^2(u''t-2u't^{-2}+2ut^{-3})+3t(u't^{-1}-ut^{-2})+ut^{-1}&=0\\
u''t+(-2+3)u'+(2-3+1)ut^{-1}&=0\\
u''t+u'&=0\intertext{let $w=u'$}  tw'+w=0 \;\; \text{first order linear ODE $\quad \Rightarrow \;\; $ integrating factor }\\
w'+\frac1t w =0 \quad \Rightarrow \mu(t) = e^{\intop \frac1t dt} = e^{\ln|t|}  = t\\
tw(t) = C \quad \Rightarrow w=C/t\\
u'=w=C/t \Rightarrow u(t)=\int C/t dt = C \ln(t) \Rightarrow \quad  \Aboxed{y_2 &= t^{-1} \ln(t)}}
}

}

\slide[ Reduction of Order: $p(t)y''+q(t)y'+r(t)y = 0$]{
\enum{
\item Given one solution $y_1(t)$, guess $y_2=u(t)y_1$ where $u(t)$ is unknown
\vfill
\item Plug $y_2$ into ODE. If you are lucky all the terms with $u(t)$ will cancel. 
\subitem{ New ODE for $u(t)$: \hfill $s(t)u''+v(t)u'=0$ \hfill  (2$^{nd}$ order $\rightarrow$ 1$^{st}$ order)\item Let $w=u'$\hfill  $ \Rightarrow $\hspace{2em}$ w'+\frac{v(t)}{s(t)}w=0$ \hfill ~ \item Integrating factor: $\mu(t) = e^{\intop \frac{v(t)}{s(t)}  dt} \;\; \Rightarrow \;\; w=C/\mu(t)$ 
\item Then $u(t) = \intop 1/\mu(t) dt$\hfill (without loss of generality, set  $C=1$) }
\vfill
\item General Solution:
\vspace{-1.25em}
\[y_h = c_1 y_1(t)+c_2y_1(t) \intop 1/\mu(t) dt\]

}
\vfill
Note: This method only works if you already have $y_1(t)$.
}
