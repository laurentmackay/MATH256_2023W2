\slide[Recall: Superposition  for Linear Homogeneous ODEs]{\vspace{-.75em}
Suppose the linearly independent functions $y_1(t)$ and $y_2(t)$ both solve the 2$^{\text{nd}}$ order \uline{linear} homogeneous ODE given by 
\algn{ &\op{y}=0,&\text{then } \quad  y_h &=c_1y_1(t) + c_2y_2(t)\intertext{is the general solution to the homogeneous ODE.}}
\vfill

Proof:
\algn{\op{c_1y_1+c_2y_2} &\overset{\text{Linearity 1}}{=}  \op{c_1y_1}+\op{c_2y_2}\\
&\overset{\text{Linearity 2}}{=} c_1\op{y_1} + c_2 \op{y_2}\\
&\quad\;\; =c_1\cdot 0 + c_2 \cdot 0  = 0
}
}

\slide[Superposition for Linear Inhomogeneous ODEs]{\vspace{-1.05em}
Suppose the linearly independent functions $y_1(t)$ and $y_2(t)$ both solve the 2$^{\text{nd}}$ order \uline{linear} homogeneous ODE given by 
\algn{ &\op{y}=0,&\text{then } \quad  y_h &=c_1y_1(t) + c_2y_2(t)\intertext{is the general solution to the homogeneous ODE. If some particular solution $y_p$ solves \vspace{-1.25em}} &\op{y_p}=g(t)\neq 0 ,&\text{then } \quad  y &=\overbrace{c_1y_1(t) + c_2y_2(t)}^{\text{complementary solution}} + y_p\intertext{is the general solution to the inhomogeneous ODE.}}

\vspace{-1.5em}
Proof:\student{
\algn{\op{c_1y_1+c_2y_2+y_p} &\overset{\text{Linearity 1}}{=}  \op{c_1y_1}+\op{c_2y_2}+\op{y_p}\\
&\overset{\text{Linearity 2}}{=}\underbrace{ c_1\op{y_1} + c_2 \op{y_2}}_{0}+\underbrace{\op{y_p}}_{g(t)} = g(t)
}}
For proof of uniqueness of $y_p$, see DiffQs \S 2.5.1
}

\slide[Solving Linear Inhomogeneous IVPs]{\vspace{-1em}
 \[\op{y(t)} = g(t)\neq0 \quad \Rightarrow \quad y(t) = y_h(t) + y_p(t) \]

\ex{Second order $\op{y} =g(t) \neq 0$ with $y(0)=y_0$, $y'(0)=v_0$ }
\student{\[y_h = c_1y_1(t)+c_2y_2(t)\]\vspace{-3em}

\algn{
y(0) = y_0 =& c_1y_1(0) + c_2y_2(0) + y_p(0) \\
y'(0) = v_0 =& c_1y_1'(0) + c_2y_2'(0) + y_p'(0) }

Convert to matrix notation:
\[ \mat{cc}{y_1(0)&y_2(0)\\y_1'(0)&y_2'(0)}\mat{c}{c_1\\c_2}=\mat{c}{y_0- y_p(0)\\v_0- y_p'(0)}\]
\vfill

\centerline{Same solvability condition as homogeneous IVPs ($W\neq0$).}\vfill
\centerline{Do not forget to include $y_p$ when solving for $c_1$ and $c_2$!}
}
}



\subsection{Method of Undetermined Coefficients }
\slide[Rough idea for finding $y_p$ for  $\op{y}=g(t)$]{
\vfill
Method of Undeteremined Coefficients:
\vfill
\enum{
\item Find the family of functional forms obtained  by differentiating $g(t)$.\vfill
\item Guess a linear combination of the family members for $y_p$.\vfill
\item Plug guess into ODE and solve the algebraic system.\vfill
}
}




\slide{\ex{ Find the particular solution to $y''+2y'+2y=2t^2-2$.}
\vfill
\student{
Guess: $y_p=At^2+Bt+C \qquad$ (3 Unknown Coefficients)
\algn{y_p'=2At+B,\quad y_p''=2A& \\
2A+2(2At+B)+2(At^2+Bt+C)&=2t^2-2\\
2At^2+ (4A+2B)t+2(A+B+C)&=2t^2-2
}
group by powers of $t$: (3 Algebraic Equations)
\algn{\uline{t^2:}& \quad 2At^2  = 2t^2 \quad \Rightarrow A=1\\
\uline{t^1:} &\quad 4\cancelto{1}{A}t + 2Bt = 0\\
&\quad  4+2B = 0 \quad \Rightarrow B=-2\\
\uline{t^0:} &\quad  2\cancelto{1}{A}+2\cancelto{-2}{B} +2C = -2\\
&\quad 2-4+2C = -2 \quad \Rightarrow C=0
&\quad \Aboxed{y_p=t^2-2t}
}

}
}


\slide{\ex{ Solve $y''+2y'+2y=2t^2-2$ with $y(0)=1, y'(0)=-3$ . }
\student{
\algn{y &= c_1 y_1 + c_2 y_2 + \underbrace{y_p}_{t^2-2t}\\
y_{1,2} &= e^{rt} & r&=\frac{-2\pm\sqrt{4-8}}{2}\\
&&&=-1\pm i\\
y(t) &=  e^{-t}\paren{c_1\cos(t) +c_2\sin(t)} +t^2-2t\intertext{initial  conditions:}
y(0)&=1=c_1 \quad \Rightarrow c_1 = 1\\
y'(t)&=-e^{t} \paren{c_1\cos(t) +c_2\sin(t)} \\&+ e^{t}  \paren{c_2\cos(t) - c_1\sin(t)}  + 2t - 2\\\\
y'(0)& = -3 = -c_1 +c_2 -2 \Rightarrow c_2 = 0\\
\Aboxed{y(t)& = e^{-t}\cos(t)+t^2-2t}
}}

\subsection{Method of Undetermined Coefficients }
\slide[ Method of Undetermined Coefficients: $\op{y}=g(t)$]{\vspace{-1.05em}
\enum{
\item Find the homogeneous solutions: \[2^{nd} \text{order }\Rightarrow\;  y_h=c_1y_1+c_2y_2\]
\item Find the family of functional forms obtained  by differentiating $g(t)$.
\ex{$g(t)=t^2e^{-t}$}
\student{
\[g'=2\uline{te^{-t}} -t^2e^{-t} \qquad g''=\uline{2e^{-t}} -2te^{-t}-2te^{-t} + t^2e^{-t}\]
\[\text{family } =\{t^2e^{-t}, te^{-t}, e^{-t} \} \]}\vspace{-1em}
\item Guess a linear combination of the family members for $y_p$.
\student{\[y_p=At^2e^{-t}+Bte^{-t}+Ce^{-t}\]}
\vspace{-2em}
\item Plug guess into ODE and solve the algebraic system.
\vfill
\student{group coeffs. by funcs of $t$, \hfill \ex{3 eqs., 3 unknowns ($A,\;B,\;C$)} }
}
}

\slide[Complication: Mathematical Resonance \hfill ($\op{y}=g(t)$)]{
\vspace{-1.25em}
\enum{
\item Find the homogeneous solutions:
$y_h=c_1y_1+c_2y_2$
\item Find the family of functional forms obtained  by differentiating $g(t)$.

\item Guess a linear combination of the family members for $y_p$.
\[ g(t) = t^2e^{-t}\Rightarrow \text{naively  guess: } y_p=At^2e^{-t}+Bte^{-t}+Ce^{-t} \]
\vspace{-2em}
\item Plug guess into ODE and solve the algebraic system.
\ex{ suppose $ y_1=e^{-t} \text{ and } y_2=e^{3t}$}
\student{
\algn{\op{y_p} = A\op{t^2e^{-t} } + B \op{te^{-t}} + \cancelto{0} {C\op{e^{-t}}} = t^2e^{-t}}
\uline{Problem:} 3 eqs, 2 unkowns ($A,\;B$) $\Rightarrow $ impossible to solve  \\ \centerline{(overdetermined system)}\vfill
\uline{Solution:} Mulitply the problematic family member by $t$ until we get a new L.I. family member that is not $y_1$ or $y_2$.
}

}

}

\vspace{-1em}

}
